### RANK

RANK() OVER([PARTITION BY COL_NAME_1] ORDER BY COL_NAME_2 [DESC])

- 기본 형식
{% highlight sql %}
SELECT
    RANK() OVER([PARTITION BY 파티션_기준_속성_목록] [ORDER BY 정렬_조건_목록]) AS 별칭
FROM
    테이블명;
{% endhighlight %}
